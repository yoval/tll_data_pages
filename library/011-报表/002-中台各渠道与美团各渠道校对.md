# 中台各渠道与美团各渠道校对

## 1. 中台数据

数据库拉取：

```sql
-- 查询时段
WITH period_table AS (
	SELECT '202501' AS start_date, '202503' AS end_date
)
SELECT
    DATE_FORMAT(day_table.pay_way_created_time, '%Y%m') AS 月份, 
    shop_id AS 门店编号,
    payment_channels AS 渠道,
    SUM(amount) AS 流水,
    SUM(discount) AS 优惠,
    SUM(income) AS 实收 
FROM
    ads_dbs_trade_shop_pay_channel_di AS day_table
CROSS JOIN
    period_table AS pt
WHERE 
    DATE_FORMAT(day_table.pay_way_created_time, '%Y%m') BETWEEN pt.start_date AND pt.end_date
GROUP BY
    DATE_FORMAT(day_table.pay_way_created_time, '%Y%m'),
    shop_id,
    payment_channels;
```

```query
    // 7. 替换空值为 0
    替换空值 = Table.TransformColumns(已透视列, List.Transform(List.RemoveItems(Table.ColumnNames(已透视列), {"门店编号&月份"}), (col) => {col, (value) => if value = null then 0 else value, type number})),
    
    // 8. 新增列：pos+小程序流水，值为 pos列 + 小程序列
    添加POS小程序流水 = Table.AddColumn(替换空值, "pos+小程序流水", each Record.Field(_, "pos") + Record.Field(_, "甜啦啦小程序"), type number),
    
    // 9. 新增列：抖音团购流水，值为 抖音团购列 + 抖音小程序列
    添加抖音团购流水 = Table.AddColumn(添加POS小程序流水, "抖音团购流水", each Record.Field(_, "抖音团购") + Record.Field(_, "抖音小程序"), type number),
    
    // 10. 新增列：美团/大众点评流水，值为 美团/大众点评团购 列 + 美团/大众点评小程序 列
    添加美团大众点评团购流水 = Table.AddColumn(添加抖音团购流水, "美团/大众点评流水", each 
        Record.Field(_, "美团/大众点评团购") + Record.Field(_, "美团/大众点评小程序"), type number),
    
    // 11. 批量重命名列
    // 将以下列名重命名：
    // - 快手团购 -> 快手团购流水
    // - 美团外卖 -> 美团外卖流水
    // - 饿了么外卖 -> 饿了么外卖流水
    重命名列 = Table.RenameColumns(添加美团大众点评团购流水,{
        {"快手团购", "快手团购流水"},
        {"美团外卖", "美团外卖流水"},
        {"饿了么外卖", "饿了么外卖流水"}
    }),
    
    // 12. 新增“营业额”列，值为 pos+小程序流水, 抖音团购流水, 美团/大众点评流水, 快手团购流水, 美团外卖流水, 饿了么外卖流水 列之和
    添加总营业额 = Table.AddColumn(重命名列, "营业额", each 
        Record.Field(_, "pos+小程序流水") + 
        Record.Field(_, "抖音团购流水") + 
        Record.Field(_, "美团/大众点评流水") + 
        Record.Field(_, "快手团购流水") + 
        Record.Field(_, "美团外卖流水") + 
        Record.Field(_, "饿了么外卖流水"), type number),

    // 13. 选择并调整列顺序，仅保留所需列并按指定顺序排列
    最终结果 = Table.SelectColumns(添加总营业额, {
        "门店编号&月份",
        "营业额",
        "pos+小程序流水",
        "抖音团购流水",
        "美团/大众点评流水",
        "快手团购流水",
        "美团外卖流水",
        "饿了么外卖流水"
    })
in
    最终结果
```

##### 3. 合并

```query
let
    // 1. Full Outer Join 连接两个表
    源 = Table.NestedJoin(中台各渠道, {"门店编号&月份"}, 美团各渠道, {"门店编号&月份"}, "美团各渠道", JoinKind.FullOuter),
    
    // 2. 展开美团各渠道的指定列（确保列名正确）
    #"展开的“美团各渠道”" = Table.ExpandTableColumn(源, "美团各渠道", 
        {"门店编号&月份", "营业额", "pos+小程序流水", "抖音团购流水", "美团/大众点评流水", "快手团购流水", "美团外卖流水", "饿了么外卖流水"}, 
        {"美团各渠道.门店编号&月份", "美团各渠道.营业额", "美团各渠道.pos+小程序流水", "美团各渠道.抖音团购流水", "美团各渠道.美团/大众点评流水", "美团各渠道.快手团购流水", "美团各渠道.美团外卖流水", "美团各渠道.饿了么外卖流水"}),
    
    // 3. 添加“表ID”列（直接引用字段名 + 空值处理）
    添加表ID = Table.AddColumn(#"展开的“美团各渠道”", "表ID", each 
        if [#"门店编号&月份"] <> null and [#"门店编号&月份"] <> "" then [#"门店编号&月份"]  // 先检查中台表字段
        else if [#"美团各渠道.门店编号&月份"] <> null and [#"美团各渠道.门店编号&月份"] <> "" then [#"美团各渠道.门店编号&月份"]  // 再检查美团表字段
        else null,  // 兜底返回 null
        type text
    )
in
    添加表ID
```
